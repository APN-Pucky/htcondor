% If we don't use math mode, the man pages and the HTML both suck less.  Note
% that using \tt{} will cause extra space to appear in the manual; we can
% probably fix this, but we haven't yet.  Some of the symbol definitions
% seem a little arbitrary, but not all symbolic commands output ASCII into
% the manual.  We can probably give this another go when we switch to building
% the manual with htlatex; we'll probably need to update makeman when we do,
% if we can't abandon it entirely.
\newcommand{\Bar}{\textbar}                         % |
\newcommand{\Dots}{\dots}                           % ...
\newcommand{\Percent}{\%{}}                         % %
\newcommand{\Tilde}{\~{}}                           % ~
\newcommand{\Circum}{\^{}}                          % ^
\newcommand{\Lbr}{[}                                % [
\newcommand{\Rbr}{]}                                % ]
\newcommand{\lt}{\textless}                         % <
\newcommand{\lte}{\textless{}=}                     % <=
\newcommand{\gt}{\textgreater}                      % >
\newcommand{\gte}{\textgreater{}=}                  % >=
\newcommand{\eq}{=}                                 % =
\newcommand{\Star}{\textasteriskcentered}           % *
